\begin{rSection}{Working Papers}
	\begin{itemize}

		\item[--] \href{https://zhifeng-cai.github.io/papers/the_ai_skill_trap.pdf}{The AI Skill Trap}, 2026
		\begin{itemize}
			\item AI raises short-run productivity but slows skill formation. A data-quality externality generates an \emph{AI skill trap}: workers become AI-dependent and fail to accumulate expertise. In a calibrated model, a constrained planner that internalizes the externality eliminates the trap and substantially raises long-run welfare
		\end{itemize}

		\item[--] \href{https://zhifeng-cai.github.io/papers/draftqje_2024.pdf}{Generative AI and Data Quality: Implications for Productivity, Labor Displacement, and Policy}, 2024
		\begin{itemize}
			\item Conditionally accepted at \emph{Review of Financial Studies}
			\item Will AI put us out of work? Not as alarming as it seems: the usefulness of AI depends on how good the data is, and adoption of AI could itself depress the quality of that data
		\end{itemize}

		\item[--] \href{https://zhifeng-cai.github.io/papers/credit_land_bubble__2_.pdf}{Financial Development and the Housing Kuznets Curve}, with \href{https://fengdongecon.weebly.com/}{Feng Dong} and Haoning Sun, 2024
		\begin{itemize}
			\item R\&R at \emph{International Economic Review}
			\item We document that the house price-to-rent ratio first increases then decreases with financial development. A simple model of financial frictions with a land sector can account for this pattern
		\end{itemize}

		\item[--] \href{https://zhifeng-cai.github.io/papers/lfp_paper__55__.pdf}{Labor Participation, Human Capital Accumulation, and the Business Cycle}, with \href{https://www.rosemarykaiser.com/}{Rosemary Kaiser}, 2024
		\begin{itemize}
			\item In a quantitative life-cycle model with endogenous labor participation, subsidies for job search among the elderly population can enhance overall welfare
		\end{itemize}

		\item[--] \href{https://zhifeng-cai.github.io/papers/cai_heathcote_gr_clean.pdf}{The Great Resignation and Optimal Unemployment Insurance}, with \href{http://www.jonathanheathcote.com/}{Jonathan Heathcote}, 2024 \href{https://cepr.org/publications/dp18526}{[CEPR WP]}
		\begin{itemize}
			\item Optimal design of unemployment insurance during the era of the Great Resignation, when many workers quit their jobs
		\end{itemize}

		\item[--] \href{https://zhifeng-cai.github.io/papers/draft0331.pdf}{Self-fulfilling Resignation: Business-Cycle and Policy Implications}, 2023
		\begin{itemize}
			\item Can people quit their jobs because they expect others are doing the same? Yes, and this has new business-cycle and policy implications
		\end{itemize}

	\end{itemize}
\end{rSection}
